//PORT detection
Start-Sleep -Seconds 6
$initialPorts = [System.IO.Ports.SerialPort]::GetPortNames()
echo "InitialPorts:" $initialPorts
while (!$newPort) {
	Start-Sleep -Milliseconds 500
	$currentPorts = [System.IO.Ports.SerialPort]::GetPortNames()
	$newPort = $currentPorts | Where-Object { $_ -notin $initialPorts }
}
echo "CurrentPorts:" $currentPorts
echo "NewPort:" $newPort
Start-Sleep -Seconds 1
$port = new-Object System.IO.Ports.SerialPort $newPort,9600,None,8,one
$port.Open()
Start-Sleep -Seconds 1
$port.WriteLine($newPort)
$port.Close()

//Get data
$portName = Read-Host
$port = New-Object System.IO.Ports.SerialPort $portName,9600,None,8,one
$port.Open()
Start-Sleep -Seconds 1
$ipAddresses = Get-CimInstance Win32_NetworkAdapterConfiguration |
Where-Object { $_.IPAddress -ne $null } |
ForEach-Object { $_.IPAddress } |
Where-Object { $_ -match '^\d{1,3}(\.\d{1,3}){3}$' }
foreach ($ip in $ipAddresses) {
	$port.WriteLine("IPv4=" + $ip)
}
$currentTime = Get-Date -Format "HH:mm:ss"
$port.WriteLine("Time=" + $currentTime)
$battery = Get-CimInstance Win32_Battery
if ($battery) {
	$port.WriteLine("Battery=" + $battery.EstimatedChargeRemaining)
} else {
	$port.WriteLine("Battery=NA")
}
$lastBoot = Get-CimInstance Win32_OperatingSystem | Select-Object -ExpandProperty LastBootUpTime
$uptime = [math]::Round(((Get-Date) - $lastBoot).TotalHours,1)
$port.WriteLine("Uptime=" + $uptime)
$cpu = (Get-CimInstance Win32_Processor | Select-Object -ExpandProperty LoadPercentage)
$port.WriteLine("CPU=" + $cpu)
$ram = Get-CimInstance Win32_OperatingSystem
$port.WriteLine("FreeRAM=" + [math]::Round($ram.FreePhysicalMemory / 1MB, 2))
$port.WriteLine("TotalRAM=" + [math]::Round($ram.TotalVisibleMemorySize / 1MB, 2))
Get-CimInstance Win32_LogicalDisk | ForEach-Object {
	$port.WriteLine("DiskID=" + $_.DeviceID)
	$port.WriteLine("FreeSpace=" + [math]::Round($_.FreeSpace / 1GB, 2))
	$port.WriteLine("TotalSize=" + [math]::Round($_.Size / 1GB, 2))
}
$port.Close()
